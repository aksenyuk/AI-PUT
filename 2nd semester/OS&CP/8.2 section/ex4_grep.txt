#include<stdio.h>
#include<string.h>

int main(int argc ,char *argv[])
{
        FILE *file;
        char line[256];
	char *check = argv[1];
        file = fopen(argv[2],"r");

        while(fgets(line, sizeof line, file)){
	  char *pos = strstr(line, check);
          if(strstr(line, check) != NULL &&  pos - line == 0) {
             printf("%s", line);
             break;
           }
         }      
     
      
        fclose(file);
        return 0;
}
